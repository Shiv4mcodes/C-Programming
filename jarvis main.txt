    def init_listener(self):

            try:

                self.recognizer.pause_threshold = 1.2 # Give user more time to pause between words
                self.recognizer.non_speaking_duration = 0.8
                self.microphone = sr.Microphone()
                self.listen_timeout = 5
                self.listen_phrase_time_limit = 15 # Increase for long commands

            except Exception as e:

        else:

    def speak(self, text, vibe=None):
        print(f"Jarvis: {text}")
        asyncio.run(self._speak_async(text, vibe))

    def detect_language(self, text: str) -> str:
        """Returns 'hi' for Hindi/Hinglish, 'en' for English."""
        import re
        HINDI_MARKERS = re.compile(
            r"\b(hai|hain|kya|bhai|yaar|kar|raha|rahi|toh|aur|nahi|nahin|"
            r"mein|main|tera|mera|karo|bata|chal|theek|arrey|sun|bol|bas)\b",
            re.IGNORECASE
        )
        words = text.split()
        hindi_hits = len(HINDI_MARKERS.findall(text))
        ratio = hindi_hits / max(len(words), 1)
        return "hi" if ratio >= 0.25 else "en"

    def get_sarvam_pace(self, vibe: dict) -> float:
        if not vibe: return 1.0
        if vibe.get("hype_mode"):      return 1.3
        elif vibe.get("banter_mode"):  return 1.2
        elif vibe.get("emotional_mode"): return 0.85
        elif vibe.get("technical_mode"): return 1.1
        else:                          return 1.0

    def get_christopher_params(self, vibe: dict) -> dict:
        if not vibe: return {"rate": "+8%",  "pitch": "+0Hz"}
        if vibe.get("hype_mode"):        return {"rate": "+30%", "pitch": "+8Hz"}
        elif vibe.get("banter_mode"):    return {"rate": "+20%", "pitch": "+4Hz"}
        elif vibe.get("emotional_mode"): return {"rate": "-8%",  "pitch": "-3Hz"}
        elif vibe.get("technical_mode"): return {"rate": "+10%", "pitch": "+0Hz"}
        else:                            return {"rate": "+8%",  "pitch": "+0Hz"}

    def strip_hindi_for_christopher(self, text: str) -> str:
        replacements = {
            "yaar": "bro",     "bhai": "bro",      "yrr": "yo",
            "arrey": "hey",    "acha": "okay",      "nahi": "no",
            "haan": "yeah",    "chal": "alright",   "mast": "sick",
            "theek": "fine",   "bata": "tell me",   "dekh": "look",
            "sun": "listen",   "bol": "say",        "sala": "dude",
            "oye": "hey",      "sahi": "right",     "ekdum": "totally",
            "bilkul": "absolutely", "bindaas": "chill"
        }
        for hindi, english in replacements.items():
            text = re.sub(rf'\b{hindi}\b', english, text, flags=re.IGNORECASE)
        return text

    async def _speak_async(self, text, vibe=None):
        self.state = "speaking"
        # Wait for any in-progress speech to finish (max 10s) before acquiring lock
        deadline = 10.0
        waited = 0.0
        while self._speak_lock.locked() and waited < deadline:
            await asyncio.sleep(0.1)
            waited += 0.1
        with self._speak_lock:
            try:
                import tempfile, pygame
                
                lang = self.detect_language(text)
                if lang == "hi" and sarvam_client:
                    pace = self.get_sarvam_pace(vibe)
                    audio = sarvam_client.text_to_speech.convert(
                        text=text,
                        target_language_code="hi-IN",
                        model="bulbul:v3",
                        speaker="rohan",
                        pace=pace,
                    )
                    raw = audio.audios[0]
                    if isinstance(raw, str):
                        import base64
                        audio_bytes = base64.b64decode(raw)
                    else:
                        audio_bytes = raw
                    with tempfile.NamedTemporaryFile(delete=False, suffix=".wav") as f:
                        tmp_path = f.name
                        f.write(audio_bytes)
                else:
                    clean_text = self.strip_hindi_for_christopher(text)
                    params = self.get_christopher_params(vibe)
                    communicate = edge_tts.Communicate(
                        text=clean_text,
                        voice="en-US-ChristopherNeural",
                        rate=params["rate"],
                        pitch=params["pitch"]
                    )
                    with tempfile.NamedTemporaryFile(delete=False, suffix=".mp3") as f:
                        tmp_path = f.name
                    await communicate.save(tmp_path)
                
                pygame.mixer.init()
                pygame.mixer.music.load(tmp_path)
                pygame.mixer.music.play()
                while pygame.mixer.music.get_busy():
                    await asyncio.sleep(0.1)
                pygame.mixer.music.unload()
                pygame.mixer.quit()
                if os.path.exists(tmp_path):
                    os.remove(tmp_path)
            except Exception as e:
                print(f"[TTS ERROR] {e}")
                communicate = edge_tts.Communicate(
                    text=text,
                    voice="en-US-ChristopherNeural",
                    rate="+8%",
                    pitch="+0Hz"
                )
                import tempfile, pygame
                with tempfile.NamedTemporaryFile(delete=False, suffix=".mp3") as f:
                    tmp_path = f.name
                await communicate.save(tmp_path)
                
                pygame.mixer.init()
                pygame.mixer.music.load(tmp_path)
                pygame.mixer.music.play()
                while pygame.mixer.music.get_busy():
                    await asyncio.sleep(0.1)
                pygame.mixer.music.unload()
                pygame.mixer.quit()
                if os.path.exists(tmp_path):
                    os.remove(tmp_path)
        self.state = "idle"

    def run(self):
        print("\n✅ Jarvis Agent V2.5 is ONLINE.")
        print("Listening for 'Hey Jarvis'...")
        try:
            while True:
                if wait_for_wake_word(self):
                    command = listen(self)
                    if command:
                        if any(w in command for w in ["exit", "quit", "goodbye", "bye"]):
                            self.speak("Goodbye, sir.")
                            break
                        self.orchestrator.handle_user_input(command)
                        # Keep awake for 5 seconds after responding
                        self.awake_until = time.time() + 5
                time.sleep(0.1)
        except KeyboardInterrupt:
            print("\nEmergency Shutdown Initiated.")
